# Icons

```js
import {IconName} from '@symphony/design-system/icons';
```

To change the color of an icon use the `color` property and set to one of: `light`, `regular`, `primary`, `error`, `gray`, `inherit`.

---

export const Icon = ({icon, name}) => {
  const classes = useStyles();
  return (
    <Grid item className={classes.iconRoot} xs={3}>
      {icon}
      <Text className={classes.iconName} variant="body1">
        {name}
      </Text>
    </Grid>
  );
};

export const IconsRoot = () => {
  const classes = useStyles();
  return (
    <div className={classes.root}>
      <Grid className={classes.iconsContainer} container>
        {Object.keys(Icons)
          .sort()
          .map(name => {
            // eslint-disable-line import/namespace
            const SvgIcon = Icons[name];
            return <Icon key={name} icon={<SvgIcon />} name={name} />;
          })}
      </Grid>
    </div>
  );
};

<IconsRoot />
